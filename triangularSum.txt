class Solution {
    public int triangularSum(int[] nums) {
        int n = nums.length;
        // Process until only one element remains
        while (n > 1) {
            for (int i = 0; i < n - 1; i++) {
                nums[i] = (nums[i] + nums[i + 1]) % 10;
            }
            n--; // shrink effective size
        }
        return nums[0]; // final result
    }
}
