class Solution {
    static int[] val = {1000,900,500,400,100,90,50,40,10,9,5,4,1};
    static String[] sym = {"M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"};

    public String intToRoman(int n) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < val.length; i++) {
            while (n >= val[i]) {
                n -= val[i];
                sb.append(sym[i]);
            }
        }
        return sb.toString();
    }
}
