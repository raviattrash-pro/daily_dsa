class Solution {
    public boolean isValid(String s) {
        char[] stack = new char[s.length()];
	    int i = 0;
	    for (char c : s.toCharArray()) {
	        if (c == '(') stack[i++] = ')';
	        else if (c == '{') stack[i++] = '}';
	        else if (c == '[') stack[i++] = ']';
	        else if (i == 0 || stack[--i] != c) return false;
	    }
	    return i == 0; 
    }
}
